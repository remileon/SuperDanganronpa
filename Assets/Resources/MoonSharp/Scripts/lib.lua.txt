function say(str, speed, auto)
    Say(str, speed, auto, 1)
    coroutine.yield()
end

function type(str, speed, auto)
    Say(str, speed, auto, 0)
    coroutine.yield()
end

function choose(str, options)
    Choose(str, options)
    idx = coroutine.yield()
    options[idx].action()
end

function curry(func, param)
    return function()
        return func(param)
    end
end
